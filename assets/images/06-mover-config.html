#!/usr/bin/env bash
# FugginNAS cache mover — generated by FugginNAS setup
CACHE=/mnt/cache
POOL=/mnt/pool
MIN_AGE_HOURS=24
MIN_FREE_PCT=20

free_pct=$(df --output=pcent "$CACHE" | tail -1 | tr -d ' %')
if (( 100 - free_pct > MIN_FREE_PCT )); then
  echo "[MOVER] Cache ${free_pct}% full — threshold not met, skipping."
  exit 0
fi
find "$CACHE" -mindepth 1 -mmin +$((MIN_AGE_HOURS * 60)) -type f \
  | while read -r f; do
    rel="${f#$CACHE/}"
    mkdir -p "$POOL/$(dirname "$rel")"
    rsync -a --remove-source-files "$f" "$POOL/$rel"
  done
